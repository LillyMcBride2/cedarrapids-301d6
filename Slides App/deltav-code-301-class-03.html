  • The main idea of these code challenges is to reinforce that you can pass any object (such as an array) into a function, make changes to that object, and it will affect the original array.Â 

    • The function's parameter may associate a different name to the object, but both the parameter and the original variable reference the same object reference in memory.

    • If you need to make changes you need to create a separate variable in the function and return it.

    • Numbers, Strings, Booleans, null, and undefined are not assigned as references. If you pass one to a function, the browser will make a separate copy in memory. Both variables are separate from one another.

function Neighborhood(obj) {
  for(let key in obj) {
    this[key] = obj[key];
  }
  this.url = `https://www.google.com/maps/place/${this.name}+Neighborhood,${this.city}`;
  this.stars = function() {
    let result ='';
    let fullStar = '<i class="fa fa-star"></i>';
    let emptyStar = '<i class="fa fa-star-o"></i>';
    let halfStar = '<i class="fa fa-star-half-o"></i>';
    result += fullStar.repeat(this.rating);
    result += this.rating % 1 !== 0 ? halfStar : '';
    result += emptyStar.repeat(5-this.rating);
    return result + ' (' + this.rating + ')';
  };
}

'use strict';

function handleSort() {
  $('#sort-by').on('change', function () {
    let sortBy = $(this).val();
    if (sortBy === 'alpha') {
      sortAlphabetical();
    }
    else if (sortBy === 'alpha-rev') {
      sortReverseAlphabetical();
    }
    else if (sortBy === 'age') {
      sortNumerical($('.person'));
    }
    else if (sortBy === 'age-rev') {
      sortReverseNumerical($('.person'));
    }
    showPeople(1);
  });
}

function sortAlphabetical() {
  $('.person').sort((a, b) => {
    let result;
    if (($(b).data('lastname').toLowerCase()) < ($(a).data('lastname').toLowerCase())) {
      result = 1;
    }
    else if (($(b).data('lastname').toLowerCase()) > ($(a).data('lastname').toLowerCase())) {
      result = -1;
    }
    else {
      if (($(b).data('firstname').toLowerCase()) < ($(a).data('firstname').toLowerCase())) {
        result = 1;
      }
      else if (($(b).data('firstname').toLowerCase()) > ($(a).data('firstname').toLowerCase())) {
        result = -1;
      }
      else {
        result = 0;
      }
    }
    return result;
  }).appendTo('#photo-gallery');
}

function sortReverseAlphabetical() {
  $('.person').sort((a, b) => {

    if (($(b).data('lastname').toLowerCase()) < ($(a).data('lastname').toLowerCase())) {
